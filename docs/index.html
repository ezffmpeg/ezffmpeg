const EZFFMPEG = require('ezffmpeg');

const ffmpeg = new EZFFMPEG({
    width: 1920,
    height: 1080
});

// Load and concatenate videos
await ffmpeg.loadMultiple([
    {
        type: 'video',
        url: 'video1.mp4',
        rotation: -90
    },
    {
        type: 'video',
        url: 'video2.mp4'
    },
    {
        type: 'text',
        text: 'Hello World',
        fontSize: 48,
        fontColor: 'white'
    }
]);

// Export the final video
await ffmpeg.export({
    outputPath: 'output.mp4'
});
